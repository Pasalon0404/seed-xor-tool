interface ToolCardProps {
  name: string
  description: string
  icon: React.ComponentType<{ className?: string }>
  isActive?: boolean
  onClick?: () => void
}

function ToolCard({ name, description, icon: Icon, isActive, onClick }: ToolCardProps) {
  return (
    <button
      onClick={onClick}
      className={`
        group relative flex items-center gap-4 w-full p-4 text-left
        border-2 transition-all duration-100
        ${isActive 
          ? "border-primary bg-primary/5" 
          : "border-border bg-card hover:border-primary/50 hover:bg-secondary"
        }
        active:translate-y-px
      `}
    >
      <div className={`
        flex items-center justify-center w-11 h-11 border-2 transition-colors
        ${isActive 
          ? "border-primary text-primary bg-primary/10" 
          : "border-border text-muted-foreground group-hover:border-primary/50 group-hover:text-foreground"
        }
      `}>
        <Icon className="w-5 h-5" />
      </div>
      <div className="flex-1 min-w-0">
        <div className="font-mono text-sm font-bold tracking-wide text-foreground">
          {name}
        </div>
        <div className="font-mono text-[11px] text-muted-foreground tracking-wide uppercase">
          {description}
        </div>
      </div>
      <ChevronRight className={`
        w-4 h-4 transition-all opacity-0 group-hover:opacity-100
        ${isActive ? "text-primary opacity-100" : "text-muted-foreground group-hover:translate-x-0.5"}
      `} />
    </button>
  )
}

interface SectionProps {
  title: string
  children: React.ReactNode
}

function Section({ title, children }: SectionProps) {
  return (
    <section className="space-y-4">
      <div className="flex items-center gap-4">
        <h2 className="font-mono text-[11px] font-black tracking-[0.25em] text-primary uppercase shrink-0">
          {title}
        </h2>
        <div className="flex-1 h-px bg-border" />
      </div>
      {children}
    </section>
  )
}

export function VaultDashboard() {
  const [passphrase, setPassphrase] = useState("")
  const [showPassphrase, setShowPassphrase] = useState(false)
  const [activeTool, setActiveTool] = useState<string | null>(null)
  const [sessionTime, setSessionTime] = useState(0)

  const seedFingerprint = "BA55E6CF"

  useEffect(() => {
    const interval = setInterval(() => {
      setSessionTime((prev) => prev + 1)
    }, 1000)
    return () => clearInterval(interval)
  }, [])

  const formatSessionTime = (seconds: number) => {
    const h = Math.floor(seconds / 3600)
    const m = Math.floor((seconds % 3600) / 60)
    const s = seconds % 60
    return `${h.toString().padStart(2, "0")}:${m.toString().padStart(2, "0")}:${s.toString().padStart(2, "0")}`
  }

  return (
    <div className="min-h-screen bg-background selection:bg-primary selection:text-primary-foreground">
      {/* Top Status Bar */}
      <div className="border-b border-border bg-muted/30">
        <div className="max-w-5xl mx-auto px-6 py-2 flex items-center justify-between">
          <div className="flex items-center gap-6 font-mono text-[10px] tracking-wider text-muted-foreground">
            <div className="flex items-center gap-1.5">
              <span className="w-1.5 h-1.5 bg-chart-2 animate-pulse" />
              <span>AIR-GAPPED</span>
            </div>
            <div className="flex items-center gap-1.5">
              <span className="w-1.5 h-1.5 bg-chart-2" />
              <span>SEED ACTIVE</span>
            </div>
          </div>
          <div className="font-mono text-[10px] tracking-wider text-muted-foreground">
            SESSION <span className="text-foreground font-medium ml-1">{formatSessionTime(sessionTime)}</span>
          </div>
        </div>
      </div>

      {/* Header */}
      <header className="border-b-2 border-border">
        <div className="max-w-5xl mx-auto px-6 py-10 text-center">
          <div className="flex items-center justify-center gap-3 mb-3">
            <Shield className="w-7 h-7 text-primary" strokeWidth={2.5} />
          </div>
          <h1 className="mb-2">
            <span className="font-mono text-4xl md:text-5xl font-black tracking-tight text-foreground">
              SAFEKEEP
            </span>
            <span className="font-mono text-4xl md:text-5xl font-black tracking-tight text-primary">
              VAULT
            </span>
          </h1>
          <p className="font-mono text-sm text-muted-foreground tracking-wide">
            Air-gapped Bitcoin signing and management device
          </p>
        </div>
      </header>

      <main className="max-w-5xl mx-auto px-6 py-8 space-y-8">
        {/* Master Seed Status */}
        <div className="border-2 border-border bg-card">
          <div className="flex flex-col sm:flex-row sm:items-center gap-5 p-5">
            <div className="flex items-center gap-5">
              <div className="relative">
                <div className="px-5 py-3 bg-primary/10 border-2 border-primary">
                  <span className="font-mono text-xl font-black tracking-[0.15em] text-primary">
                    {seedFingerprint}
                  </span>
                </div>
                <div className="absolute -top-1 -right-1 w-2.5 h-2.5 bg-chart-2 border border-background" />
              </div>
              <div>
                <div className="font-mono text-sm font-bold text-foreground tracking-wide">
                  Master Seed Active
                </div>
                <div className="font-mono text-xs text-muted-foreground tracking-wide">
                  12 words loaded into RAM
                </div>
              </div>
            </div>
          </div>
        </div>

        {/* Passphrase Input */}
        <div className="border-2 border-border bg-card">
          <div className="p-5 space-y-4">
            <div className="flex flex-col lg:flex-row lg:items-center gap-4">
              <label className="font-mono text-xs font-bold text-foreground tracking-wider uppercase shrink-0 lg:w-28">
                Passphrase
              </label>
              <div className="flex-1 flex flex-col sm:flex-row gap-3">
                <div className="flex-1 relative">
                  <input
                    type={showPassphrase ? "text" : "password"}
                    value={passphrase}
                    onChange={(e) => setPassphrase(e.target.value)}
                    placeholder="Optional — type to derive a different wallet"
                    className="w-full px-4 py-3.5 bg-input border-2 border-border font-mono text-sm text-foreground placeholder:text-muted-foreground/60 focus:outline-none focus:border-primary transition-colors"
                  />
                </div>
                <div className="flex gap-2">
                  <button
                    onClick={() => setShowPassphrase(!showPassphrase)}
                    className="flex items-center justify-center gap-2 px-5 py-3.5 border-2 border-border bg-secondary font-mono text-xs font-bold text-foreground tracking-wider hover:bg-muted hover:border-foreground/30 transition-all active:translate-y-px"
                  >
                    {showPassphrase ? <EyeOff className="w-3.5 h-3.5" /> : <Eye className="w-3.5 h-3.5" />}
                    <span>{showPassphrase ? "Hide" : "Show"}</span>
                  </button>
                  <button className="flex items-center justify-center gap-2 px-5 py-3.5 border-2 border-primary bg-transparent font-mono text-xs font-bold text-primary tracking-wider hover:bg-primary hover:text-primary-foreground transition-all active:translate-y-px">

              </p>
            </div>
          </div>
        </div>

        {/* Transaction Tools */}
        <Section title="Transaction">
          <div className="grid grid-cols-1 md:grid-cols-2 gap-3">
            {transactionTools.map((tool) => (
              <ToolCard
                key={tool.id}
                name={tool.name}
                description={tool.description}
                icon={tool.icon}
                isActive={activeTool === tool.id}
                onClick={() => setActiveTool(activeTool === tool.id ? null : tool.id)}
              />
            ))}
          </div>
        </Section>

        {/* Seed Management */}
        <Section title="Seed Management">
          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-3">
            {seedManagementTools.map((tool) => (
              <ToolCard
                key={tool.id}
                name={tool.name}
                description={tool.description}
                icon={tool.icon}
                isActive={activeTool === tool.id}
                onClick={() => setActiveTool(activeTool === tool.id ? null : tool.id)}
              />
            ))}
          </div>
        </Section>

        {/* Backup & Verification */}
        <Section title="Backup & Verification">
          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-3">
            {backupTools.map((tool) => (
              <ToolCard
                key={tool.id}
                name={tool.name}
                description={tool.description}
                icon={tool.icon}
                isActive={activeTool === tool.id}
                onClick={() => setActiveTool(activeTool === tool.id ? null : tool.id)}
              />
            ))}
          </div>
        </Section>

        {/* Workspace */}
        <Section title="Workspace">
          <div className="grid grid-cols-1 md:grid-cols-2 gap-3">
            {workspaceTools.map((tool) => (
              <ToolCard

                onClick={() => setActiveTool(activeTool === tool.id ? null : tool.id)}
              />
            ))}
          </div>
        </Section>

        {/* Device Settings */}
        <div className="pt-6 border-t border-border">
          <button className="flex items-center gap-3 font-mono text-sm text-muted-foreground hover:text-foreground transition-colors group">
            <ChevronRight className="w-4 h-4 transition-transform group-hover:translate-x-0.5" />
            <Settings className="w-4 h-4" />
            <span className="tracking-wide">Device Settings</span>
          </button>
        </div>
      </main>

      {/* Footer */}
      <footer className="border-t-2 border-border mt-8">
        <div className="max-w-5xl mx-auto px-6 py-5">
          <div className="flex flex-wrap items-center justify-between gap-4 font-mono text-[10px] tracking-wider text-muted-foreground">
            <div className="flex items-center gap-8">
              <span>SAFEKEEP VAULT</span>
              <span className="text-foreground">v2.1.0</span>
            </div>
            <div className="flex items-center gap-2">
              <span className="w-1.5 h-1.5 bg-chart-2" />
              <span>INTEGRITY VERIFIED</span>
            </div>
          </div>
        </div>
      </footer>
    </div>
  )
}

export default VaultDashboard
